  <script>
    let state = { countC: 0, countJ: 0, coeffC: 0.4, coeffJ: 0.4, historyC: [], historyJ: [], syncId: "" };
    const apiBase = "https://api.npoint.io/";
    let autoSaveTimeout;
    let isSaving = false;

    function loadData() {
      const saved = localStorage.getItem('classic_2026_v9');
      if (saved) {
        state = JSON.parse(saved);
        updateCoeffVisuals('C', state.coeffC);
        updateCoeffVisuals('J', state.coeffJ);
      }
      updateUI();
    }

    function saveData() {
      state.syncId = document.getElementById('input-sync-id').value.trim();
      localStorage.setItem('classic_2026_v9', JSON.stringify(state));
      triggerAutoSave();
    }

    async function triggerAutoSave() {
      if (!state.syncId) return;
      const status = document.getElementById('cloud-status');
      clearTimeout(autoSaveTimeout);
      autoSaveTimeout = setTimeout(async () => {
        if (isSaving) return;
        isSaving = true;
        status.textContent = "Сохранение...";
        try {
          const response = await fetch(apiBase + state.syncId, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(state)
          });
          if (response.ok) status.textContent = "Сохранено";
          else status.textContent = "Ошибка сервера";
        } catch (e) {
          status.textContent = "Ошибка сети";
        } finally {
          isSaving = false;
        }
      }, 500);
    }

    function updateUI() {
      document.getElementById('count-c').textContent = state.countC;
      document.getElementById('count-j').textContent = state.countJ;
      document.getElementById('input-c').value = state.countC;
      document.getElementById('input-j').value = state.countJ;
      document.getElementById('input-sync-id').value = state.syncId;
      document.getElementById('total-clicks').textContent = state.countC + state.countJ;
      let totalScore = (state.historyC.reduce((a,b)=>a+b,0)) + (state.historyJ.reduce((a,b)=>a+b,0));
      document.getElementById('accumulated-output').textContent = totalScore.toFixed(1);
    }

    document.getElementById('btn-c').addEventListener('click', () => { state.countC++; state.historyC.push(state.coeffC); updateUI(); saveData(); });
    document.getElementById('btn-j').addEventListener('click', () => { state.countJ++; state.historyJ.push(state.coeffJ); updateUI(); saveData(); });

    function setCoeff(type, val, btn) {
      if (type === 'C') state.coeffC = val; else state.coeffJ = val;
      btn.parentElement.querySelectorAll('.seg-btn').forEach(b => b.classList.remove('active'));
      btn.classList.add('active');
      saveData();
    }

    function applyManualOverride() {
      state.countC = parseInt(document.getElementById('input-c').value) || 0;
      state.countJ = parseInt(document.getElementById('input-j').value) || 0;
      updateUI();
      saveData();
      toggleTab('manual');
    }

    function toggleTab(tab) {
      const pManual = document.getElementById('panel-manual');
      const pCloud = document.getElementById('panel-cloud');
      if (tab === 'manual') { 
        pManual.style.display = pManual.style.display === 'block' ? 'none' : 'block'; 
        pCloud.style.display = 'none'; 
      } else { 
        pCloud.style.display = pCloud.style.display === 'block' ? 'none' : 'block'; 
        pManual.style.display = 'none'; 
      }
    }

    async function cloudPull() {
      const id = document.getElementById('input-sync-id').value.trim();
      if (!id) return;
      try {
        const response = await fetch(apiBase + id);
        if (response.ok) {
          state = await response.json();
          saveData();
          updateCoeffVisuals('C', state.coeffC);
          updateCoeffVisuals('J', state.coeffJ);
          updateUI();
          alert("Данные загружены!");
        }
      } catch (e) { alert("Ошибка загрузки"); }
    }

    function updateCoeffVisuals(type, val) {
      const containerId = type === 'C' ? 'coeffs-c' : 'coeffs-j';
      document.querySelectorAll(`#${containerId} .seg-btn`).forEach(btn => {
        if (parseFloat(btn.textContent) === val) btn.classList.add('active');
        else btn.classList.remove('active');
      });
    }

    window.onload = loadData;
  </script>
